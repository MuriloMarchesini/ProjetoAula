        var cart = [
            { name: "Produto 1", price: 4300 },
            { name: "Produto 2", price: 200 },
            { name: "Produto 3", price: 300 }
        ];
        var shipping = 75;

        function displayCart() {
            var cartDiv = document.getElementById("cart");
            var total = 0;
            cart.forEach(function(item) {
                var itemDiv = document.createElement("div");
                itemDiv.className = "cart-item";
                itemDiv.innerText = item.name + " - " + "R$" + item.price;
                cartDiv.appendChild(itemDiv);
                total += item.price;
            });
            total += shipping;
            document.getElementById("total").innerText = "Total (incluindo frete): R$" + total;
        }

        function checkout() {
            alert("Compra finalizada! Obrigado pela sua compra.");
        }

        displayCart();
